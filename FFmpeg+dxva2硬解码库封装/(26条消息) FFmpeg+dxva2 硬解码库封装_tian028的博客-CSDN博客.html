  422. int VideoFFmpegDecode_dxva::dxva2_create_decoder(AVCodecContext *s)
  423. {
  424. 	dxva_context *dxva_ctx = (dxva_context *)s->hwaccel_context;
  425. 	GUID *guid_list = NULL;
  426. 	unsigned guid_count = 0, i, j;
  427. 	GUID device_guid = GUID_NULL;
  428. 	D3DFORMAT target_format = D3DFMT_UNKNOWN;
  429. 	DXVA2_VideoDesc desc = { 0 };
  430. 	DXVA2_ConfigPictureDecode config;
  431. 	HRESULT hr;
  432. 	int surface_alignment;
  433. 	int ret;
  434.  
  435. 	hr = ctx->decoder_service->GetDecoderDeviceGuids(&guid_count, &guid_list);
  436. 	if (FAILED(hr)) {
  437. 		dxva2_destroy_decoder(s);
  438. 		return AVERROR(EINVAL);
  439. 	}
  440.  
  441. 	for (i = 0; dxva2_modes[i].guid; i++) {
  442. 		D3DFORMAT *target_list = NULL;
  443. 		unsigned target_count = 0;
  444. 		const dxva2_mode *mode = &dxva2_modes[i];
  445. 		if (mode->codec != s->codec_id)
  446. 			continue;
  447.  
  448. 		for (j = 0; j < guid_count; j++) {
  449. 			if (IsEqualGUID(*mode->guid, guid_list[j]))
  450. 				break;
  451. 		}
  452. 		if (j == guid_count)
  453. 			continue;
  454.  
  455. 		//hr = IDirectXVideoDecoderService_GetDecoderRenderTargets(ctx->decoder_service, mode->guid, &target_count, &target_list);
  456. 		hr = ctx->decoder_service->GetDecoderRenderTargets(*mode->guid, &target_count, &target_list);
  457. 		if (FAILED(hr)) {
  458. 			continue;
  459. 		}
  460.  
  461. 		for (unsigned j = 0; j < target_count; j++)
  462. 		{
  463. 			const D3DFORMAT f = target_list[j];
  464. 			const d3d_format_t *format = D3dFindFormat(f);
  465. 			if (format) {
  466. 				printf("%s is supported for output\n", format->name);
  467. 			}
  468. 			else {
  469. 				printf("%d is supported for output (%4.4s)\n", f, (const char*)&f);
  470. 			}
  471. 		}
  472.  
  473. 		for (j = 0; j < target_count; j++) {
  474. 			const D3DFORMAT format = target_list[j];
  475. 			if (format == MKTAG('N', 'V', '1', '2')) {
  476. 				target_format = format;
  477. 				break;
  478. 			}
  479. 		}
  480. 		CoTaskMemFree(target_list);
  481. 		if (target_format) {
  482. 			device_guid = *mode->guid;
  483. 			break;
  484. 		}
  485. 	}
  486. 	CoTaskMemFree(guid_list);
  487. 	if (IsEqualGUID(device_guid, GUID_NULL)) {
  488. 		dxva2_destroy_decoder(s);
  489. 		return AVERROR(EINVAL);
  490. 	}
  491.  
  492. 	desc.SampleWidth = s->coded_width;
  493. 	desc.SampleHeight = s->coded_height;
  494. 	desc.Format = target_format;
  495.  
  496. 	ret = dxva2_get_decoder_configuration(s, &device_guid, &desc, &config);
  497. 	if (ret < 0) {
  498. 		dxva2_destroy_decoder(s);
  499. 		return AVERROR(EINVAL);
  500. 	}
  501. 	if (s->codec_id == AV_CODEC_ID_MPEG2VIDEO)
  502. 		surface_alignment = 32;
  503. 	else if (s->codec_id == AV_CODEC_ID_HEVC)
  504. 		surface_alignment = 128;
  505. 	else
  506. 		surface_alignment = 16;
  507. 	ctx->num_surfaces = 4;
  508. 	if (s->codec_id == AV_CODEC_ID_H264 || s->codec_id == AV_CODEC_ID_HEVC)
  509. 		ctx->num_surfaces += 16;
  510. 	else if (s->codec_id == AV_CODEC_ID_VP9)
  511. 		ctx->num_surfaces += 8;
  512. 	else
  513. 		ctx->num_surfaces += 2;
  514. 	if (s->active_thread_type & FF_THREAD_FRAME)
  515. 		ctx->num_surfaces += s->thread_count;
  516. 	ctx->surfaces = (LPDIRECT3DSURFACE9 *)av_mallocz(ctx->num_surfaces * sizeof(*ctx->surfaces));
  517. 	ctx->surface_infos = (surface_info *)av_mallocz(ctx->num_surfaces * sizeof(*ctx->surface_infos));
  518.  
  519. 	if (!ctx->surfaces || !ctx->surface_infos) {
  520. 		dxva2_destroy_decoder(s);
  521. 		return AVERROR(EINVAL);
  522. 	}
  523. 	// s->coded_width s->coded_height
  524. 	hr = ctx->decoder_service->CreateSurface(FFALIGN(s->coded_width, surface_alignment),
  525. 		FFALIGN(s->coded_height, surface_alignment),
  526. 		ctx->num_surfaces - 1,
  527. 		target_format, D3DPOOL_DEFAULT, 0,
  528. 		DXVA2_VideoDecoderRenderTarget,
  529. 		ctx->surfaces, NULL);
  530. 	if (FAILED(hr)) {
  531. 		dxva2_destroy_decoder(s);
  532. 		return AVERROR(EINVAL);
  533. 	}
  534. 	hr = ctx->decoder_service->CreateVideoDecoder(device_guid,
  535. 		&desc, &config, ctx->surfaces,
  536. 		ctx->num_surfaces, &ctx->decoder);
  537. 	if (FAILED(hr)) {
  538. 		dxva2_destroy_decoder(s);
  539. 		return AVERROR(EINVAL);
  540. 	}
  541. 	ctx->decoder_guid = device_guid;
  542. 	ctx->decoder_config = config;
  543.  
  544. 	dxva_ctx->cfg = &ctx->decoder_config;
  545. 	dxva_ctx->decoder = ctx->decoder;
  546. 	dxva_ctx->surface = ctx->surfaces;
  547. 	dxva_ctx->surface_count = ctx->num_surfaces;
  548. 	if (IsEqualGUID(ctx->decoder_guid, DXVADDI_Intel_ModeH264_E))
  549. 		dxva_ctx->workaround |= FF_DXVA2_WORKAROUND_INTEL_CLEARVIDEO;
  550. 	return 0;

  561. int VideoFFmpegDecode_dxva::dxva2_get_decoder_configuration(AVCodecContext *s, const GUID *device_guid,
  562. 	const DXVA2_VideoDesc *desc,
  563. 	DXVA2_ConfigPictureDecode *config)
  564. {
  565. 	unsigned cfg_count = 0, best_score = 0;
  566. 	DXVA2_ConfigPictureDecode *cfg_list = NULL;
  567. 	DXVA2_ConfigPictureDecode best_cfg = { { 0 } };
  568. 	HRESULT hr;
  569. 	int i;
  570.  
  571. 	//hr = IDirectXVideoDecoderService_GetDecoderConfigurations(ctx->decoder_service, device_guid, desc, NULL, &cfg_count, &cfg_list);
  572. 	hr = ctx->decoder_service->GetDecoderConfigurations(*device_guid, desc, NULL, &cfg_count, &cfg_list);
  573. 	if (FAILED(hr)) {
  574. 		return AVERROR(EINVAL);
  575. 	}
  576.  
  577. 	for (i = 0; i < cfg_count; i++) {
  578. 		DXVA2_ConfigPictureDecode *cfg = &cfg_list[i];
  579.  
  580. 		unsigned score;
  581. 		if (cfg->ConfigBitstreamRaw == 1)
  582. 			score = 1;
  583. 		else if (s->codec_id == AV_CODEC_ID_H264 && cfg->ConfigBitstreamRaw == 2)
  584. 			score = 2;
  585. 		else
  586. 			continue;
  587. 		if (IsEqualGUID(cfg->guidConfigBitstreamEncryption, DXVA2_NoEncrypt))
  588. 			score += 16;
  589. 		if (score > best_score) {
  590. 			best_score = score;
  591. 			best_cfg = *cfg;
  592. 		}
  593. 	}
  594. 	CoTaskMemFree(cfg_list);
  595. 	if (!best_score) {
  596. 		return AVERROR(EINVAL);
  597. 	}
  598. 	*config = best_cfg;
  599. 	return 0;
  600. }
