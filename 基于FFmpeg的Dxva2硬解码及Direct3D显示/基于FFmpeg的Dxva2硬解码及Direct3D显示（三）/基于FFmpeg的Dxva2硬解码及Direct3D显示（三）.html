Direct3D渲染可以通过Surface和Texture来实现，后者比较复杂，只是显示视频的化用Surface足以。Direct3D有几个重要的概念需要说明以下：

Device：就是渲染设备实例

Resources：可以理解成一个参数结构体，描述怎样渲染的细节。

Swap Chains：交换链，由一个或者多个缓冲区组成，可以理解成包含多个缓冲区的数组，如下图所示。

创建Direct3D物理设备对象实例

这一步为了得到IDirect3D9* 实例 m_pD3d9;

  1. m_hD3dDll = LoadLibrary("d3d9.dll");
  2. pDirect3DCreate9 *createD3D = (pDirect3DCreate9 *)GetProcAddress(m_hD3dDll, "Direct3DCreate9");
  4. m_pD3d9 = createD3D(D3D_SDK_VERSION);// 怎样指定GPU？

创建Direct3D渲染设备实例

这一步为了得到IDirect3DDevice9* 实例 m_pD3d9Dev;

  1. // 获取显示适配器参数
  2. D3DDISPLAYMODE displayModel;
  3. if (m_pD3d9->GetAdapterDisplayMode(D3DADAPTER_DEFAULT, &displayModel) < 0)
  4. {
  5.     return FALSE;
  6. }
  8. D3DPRESENT_PARAMETERS presentParms = { 0 };
  9. presentParms.Windowed = TRUE;			// 是否窗口显示
  10. presentParms.BackBufferCount = 0;		// 后台缓冲表面数量，指定为0时会被视为1
  11. presentParms.hDeviceWindow = m_hWnd;	// 显示窗口句柄
  12. presentParms.SwapEffect = D3DSWAPEFFECT_DISCARD;	// 交换链设置，后台缓冲使用后直接丢弃
  13. presentParms.BackBufferFormat = displayModel.Format;// 后台缓冲表面格式
  14. presentParms.EnableAutoDepthStencil = FALSE;		// 是否自动创建深度模板缓冲，适用于3D场景
  15. presentParms.Flags = D3DPRESENTFLAG_VIDEO;			// 附加特性，显示视频
  16. presentParms.FullScreen_RefreshRateInHz = D3DPRESENT_RATE_DEFAULT;	// 刷新率，窗口模式时必须为0
  17. presentParms.PresentationInterval = D3DPRESENT_INTERVAL_DEFAULT;	// 前后台缓冲区的交换频率
  19. // 查询设备是否支持硬件加速渲染
  20. DWORD behaviorFlags = D3DCREATE_SOFTWARE_VERTEXPROCESSING | D3DCREATE_MULTITHREADED;
  21. D3DDEVTYPE devType = D3DDEVTYPE_HAL;
  22. D3DCAPS9 caps;
  24. if (m_pD3d9->GetDeviceCaps(D3DADAPTER_DEFAULT, devType, &caps) >= 0)
  25. {
  26.     if (caps.DevCaps & D3DDEVCAPS_HWTRANSFORMANDLIGHT)
  27.     {
  28.         behaviorFlags = D3DCREATE_HARDWARE_VERTEXPROCESSING | D3DCREATE_MULTITHREADED | D3DCREATE_FPU_PRESERVE;
  29.     }
  30.     else
  31.     {
  32.         behaviorFlags = D3DCREATE_SOFTWARE_VERTEXPROCESSING | D3DCREATE_MULTITHREADED | D3DCREATE_FPU_PRESERVE;
  33.     }
  34. }
  35. else
  36. {
  37.     return FALSE;
  38. }
  40. // 创建渲染设备实例
  41. if (m_pD3d9->CreateDevice(D3DADAPTER_DEFAULT, devType, m_hWnd,
  42.                           behaviorFlags, &presentParms, &m_pD3d9Dev) < 0)
  43. {
  44.     return FALSE;
  45. }

创建Direct3D视频解码服务

这一步为了得到IDirectXVideoDecoderService* 实例 m_pDecoderService;

  1. m_hDxva2Dll = LoadLibrary("dxva2.dll");
  2. if (!m_hDxva2Dll)
  3. {
  4.     return FALSE;
  5. }
  7. pCreateDeviceManager9 *createDeviceManager = (pCreateDeviceManager9 *)GetProcAddress(m_hDxva2Dll, "DXVA2CreateDirect3DDeviceManager9");
  9. // 设备管理器实例标识
  10. unsigned resetToken = 0;
  11. if (createDeviceManager(&resetToken, &m_pD3d9DevMgr) < 0)
  12. {
  13.     return FALSE;
  14. }
  16. // 重置渲染设备
  17. HRESULT hr = m_pD3d9DevMgr->ResetDevice(m_pD3d9Dev, resetToken);
  18. if (FAILED(hr))
  19. {
  20.     return FALSE;
  21. }
  23. // 打开渲染设备
  24. hr = m_pD3d9DevMgr->OpenDeviceHandle(&m_hDev);
  25. if (FAILED(hr))
  26. {
  27.     return FALSE;
  28. }
  30. // 获取视频渲染服务实例
  31. hr = m_pD3d9DevMgr->GetVideoService(m_hDev, IID_IDirectXVideoDecoderService, (void **)&m_pDecoderService);
  32. if (FAILED(hr))
  33. {
  34.     return FALSE;
  35. }
